\documentclass[border=1cm]{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[scale=1.2]
% Координатные оси (серые)
\draw[gray!50, ->] (-1.2,0) -- (2.2,0) node[below, black] {$x$};
\draw[gray!50, ->] (0,-2.2) -- (0,5.2) node[left, black] {$y$};
% Деления и подписи на осях
\foreach \x in {-1,-0.5,0,0.5,1,1.5,2}
    \draw[gray!50] (\x,0.05) -- (\x,-0.05) node[below, black] {\small\x};
\foreach \y in {-2,-1,0,1,2,3,4,5}
    \draw[gray!50] (0.05,\y) -- (-0.05,\y) node[left, black] {\small\y};

% Экспонента y = e^x (синий)
\draw[domain=-1:1.44, smooth, variable=\x, blue, thick] 
    plot ({\x}, {exp(\x)});

% Натуральный логарифм y = ln(x) (черный)
\draw[domain=0.02:2, smooth, variable=\x, black, thick] 
    plot ({\x}, {ln(\x)});

% Подписи функций
\node[blue, anchor=west] at (1.2, {exp(1.2)}) {$y = e^x$};
\node[black, anchor=west] at (1.5, {ln(1.5)}) {$y = \ln x$};

\end{tikzpicture}
\end{document}
